Table users {
  id integer [primary key]
  name varchar
  email varchar
  email_verified_at timestamp
  password varchar
  remember_token varchar
  type varchar //2 valori employer si employee si admin
  created_at timestamp
  updated_at timestamp
}

Table job_domains {
  id integer [primary key]
  name varchar
  description text
  created_at timestamp
  updated_at timestamp
}

Table experience {
  id integer [primary key]
  name varchar
  min_experience_interval integer
  max_experiente_interval integer
  created_at timestamp
  updated_at timestamp
}

Table work_preferences {
  id integer [primary key]
  name varchar
  created_at timestamp
  updated_at timestamp
}

Table languages {
  id integer [primary key]
  name varchar
  created_at timestamp
  updated_at timestamp
}

Table benefits {
  id integer [primary key]
  name varchar
  created_at timestamp
  updated_at timestamp
}

Table benefit_job {
  id integer [primary key]
  benefit_id integer
  job_id integer
  created_at timestamp
  updated_at timestamp
}


Table jobs {
  id integer [primary key]
  name varchar
  description text
  employer_id integer
  domain_id integer
  experience_id integer
  work_preferences_id integer
  language_id integer
  created_at timestamp
  updated_at timestamp
  expired_at timestamp
}

Ref: jobs.experience_id > experience.id

Ref: jobs.language_id > languages.id
